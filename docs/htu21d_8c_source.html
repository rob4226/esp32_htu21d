43int htu21d_init(i2c_port_t port, int sda_pin, int scl_pin, gpio_pullup_t sda_internal_pullup, gpio_pullup_t scl_internal_pullup)
44{
45 esp_err_t ret;
46 _port = port;
47
48 // setup i2c controller
49 i2c_config_t conf = {0};
50 conf.mode = I2C_MODE_MASTER;
51 conf.sda_io_num = sda_pin;
52 conf.scl_io_num = scl_pin;
53 conf.sda_pullup_en = sda_internal_pullup;
54 conf.scl_pullup_en = scl_internal_pullup;
55 conf.master.clk_speed = 100000;
56 ret = i2c_param_config(port, &conf);
57 if (ret != ESP_OK) {
58 ESP_LOGE(TAG, "Failed to configure I2C (port %d, sda_pin %d, scl_pin %d): %s", port, sda_pin, scl_pin, esp_err_to_name(ret));
59 return HTU21D_ERR_CONFIG;
60 }
61
62 // install the driver
63 ret = i2c_driver_install(port, I2C_MODE_MASTER, 0, 0, 0);
64 if (ret != ESP_OK) {
65 ESP_LOGE(TAG, "Failed to install I2C driver: %s", esp_err_to_name(ret));
66 return HTU21D_ERR_INSTALL;
67 }
68
69 // verify if a sensor is present
70 i2c_cmd_handle_t cmd = i2c_cmd_link_create();
71 if (cmd == NULL) {
72 ESP_LOGE(TAG, "Not enough dynamic memory");
73 return HTU21D_ERR_FAIL;
74 }
75 ESP_ERROR_CHECK_WITHOUT_ABORT(i2c_master_start(cmd));
76 ESP_ERROR_CHECK_WITHOUT_ABORT(i2c_master_write_byte(
77 cmd, (HTU21D_ADDR << 1) | I2C_MASTER_WRITE, true));
78 ESP_ERROR_CHECK_WITHOUT_ABORT(i2c_master_stop(cmd));
79 ret = i2c_master_cmd_begin(port, cmd, 1000 / portTICK_PERIOD_MS);
80 if (ret != ESP_OK) {
81 ESP_LOGE(TAG, "HTU21D sensor not found on bus: %s", esp_err_to_name(ret));
82 return HTU21D_ERR_NOTFOUND;
83 }
84 ESP_LOGI(TAG, "HTU21D sensor initialized successfully.");
85
86 // Per datasheet, it is recommended to soft reset the HTU21D sensor on start:
87 ret = htu21d_soft_reset();
88 if (ret != HTU21D_ERR_OK) {
89 ESP_LOGE(TAG, "Failed to soft reset the HTU21D sensor after initializing it, error: 0x%02X", ret);
90 return ret;
91 }
92
93 return HTU21D_ERR_OK;
94}

101float htu21d_read_temperature()
102{
103 // get the raw value from the sensor
104 uint16_t raw_temperature = read_value(TRIGGER_TEMP_MEASURE_NOHOLD);
105 if (raw_temperature == 0) {
106 return -999;
107 }
108
109 // return the real value, formula in datasheet
110 return (raw_temperature * 175.72 / 65536.0) - 46.85;

160float htu21d_compute_partial_pressure(float temperature)
161{
162 return pow(10, HTU21_CONSTANT_A - HTU21_CONSTANT_B / (temperature + HTU21_CONSTANT_C));
163}

186float htu21d_compute_dew_point(float temperature, float relative_humidity)
187{
188 float partial_pressure = htu21d_compute_partial_pressure(temperature);
189
190 return - HTU21_CONSTANT_B /
191 (log10(relative_humidity * partial_pressure / 100.0F) - HTU21_CONSTANT_A)
192 - HTU21_CONSTANT_C;
193}

235int htu21d_soft_reset()
236{
237 esp_err_t ret;
238
239 // send the command
240 i2c_cmd_handle_t cmd = i2c_cmd_link_create();
241 if (cmd == NULL) {
242 ESP_LOGE(TAG, "Not enough dynamic memory");
243 return HTU21D_ERR_FAIL;
244 }
245 ESP_ERROR_CHECK_WITHOUT_ABORT(i2c_master_start(cmd));
246 ESP_ERROR_CHECK_WITHOUT_ABORT(i2c_master_write_byte(
247 cmd, (HTU21D_ADDR << 1) | I2C_MASTER_WRITE, true));
248 ESP_ERROR_CHECK_WITHOUT_ABORT(i2c_master_write_byte(cmd, SOFT_RESET, true));
249 ESP_ERROR_CHECK_WITHOUT_ABORT(i2c_master_stop(cmd));
250 ret = i2c_master_cmd_begin(_port, cmd, 1000 / portTICK_PERIOD_MS);
251 ESP_ERROR_CHECK_WITHOUT_ABORT(ret);
252 i2c_cmd_link_delete(cmd);
253
254 switch (ret) {
255
256 case ESP_ERR_INVALID_ARG:
257 ESP_LOGE(TAG, "Soft reset failed, parameter error.");
258 return HTU21D_ERR_INVALID_ARG;
259
260 case ESP_FAIL:
261 ESP_LOGE(TAG, "Sending Soft Reset command error, the HTU21D slave hasn't ACK the transfer.");
262 return HTU21D_ERR_FAIL;
263
264 case ESP_ERR_INVALID_STATE:
265 ESP_LOGE(TAG, "Soft reset failed, I2C driver not installed or not in master mode.");
266 return HTU21D_ERR_INVALID_STATE;
267
268 case ESP_ERR_TIMEOUT:
269 ESP_LOGE(TAG, "Soft reset failed, operation timeout because the I2C bus is busy.");
270 return HTU21D_ERR_TIMEOUT;
271 }
272
273 vTaskDelay(pdMS_TO_TICKS(HTU21_RESET_TIME));
274
275 ESP_LOGI(TAG, "HTU21D sensor soft reset was successful.");
276
277 return HTU21D_ERR_OK;
278}

361uint16_t read_value(uint8_t command)
362{
363 esp_err_t ret;
364
365 // send the command
366 i2c_cmd_handle_t cmd = i2c_cmd_link_create();
367 if (cmd == NULL) {
368 ESP_LOGE(TAG, "Not enough dynamic memory");
369 return 0;
370 }
371 ESP_ERROR_CHECK_WITHOUT_ABORT(i2c_master_start(cmd));
372 ESP_ERROR_CHECK_WITHOUT_ABORT(
373 i2c_master_write_byte(cmd, (HTU21D_ADDR << 1) | I2C_MASTER_WRITE, true));
374 ESP_ERROR_CHECK_WITHOUT_ABORT(i2c_master_write_byte(cmd, command, true));
375 ESP_ERROR_CHECK_WITHOUT_ABORT(i2c_master_stop(cmd));
376 ret = i2c_master_cmd_begin(_port, cmd, 1000 / portTICK_PERIOD_MS);
377 ESP_ERROR_CHECK_WITHOUT_ABORT(ret);
378 i2c_cmd_link_delete(cmd);
379 if (ret != ESP_OK) {
380 return 0;
381 }
382
383 // wait for the sensor (50ms)
384 vTaskDelay(50 / portTICK_PERIOD_MS);
385
386 // receive the answer
387 uint8_t msb, lsb, crc;
388 cmd = i2c_cmd_link_create();
389 if (cmd == NULL) {
390 ESP_LOGE(TAG, "Not enough dynamic memory");
391 return 0;
392 }
393 ESP_ERROR_CHECK_WITHOUT_ABORT(i2c_master_start(cmd));
394 ESP_ERROR_CHECK_WITHOUT_ABORT(
395 i2c_master_write_byte(cmd, (HTU21D_ADDR << 1) | I2C_MASTER_READ, true));
396 ESP_ERROR_CHECK_WITHOUT_ABORT(i2c_master_read_byte(cmd, &msb, 0x00));
397 ESP_ERROR_CHECK_WITHOUT_ABORT(i2c_master_read_byte(cmd, &lsb, 0x00));
398 ESP_ERROR_CHECK_WITHOUT_ABORT(i2c_master_read_byte(cmd, &crc, 0x01));
399 ESP_ERROR_CHECK_WITHOUT_ABORT(i2c_master_stop(cmd));
400 ret = i2c_master_cmd_begin(_port, cmd, 1000 / portTICK_PERIOD_MS);
401 ESP_ERROR_CHECK_WITHOUT_ABORT(ret);
402 i2c_cmd_link_delete(cmd);
403 if (ret != ESP_OK) {
404 return 0;
405 }
406
407 uint16_t raw_value = ((uint16_t) msb << 8) | (uint16_t) lsb;
408 if (!is_crc_valid(raw_value, crc)) {
409 ESP_LOGE(TAG, "CRC is invalid.");
410 }
411 return raw_value & 0xFFFC;
412}

414// verify the CRC, algorithm in the datasheet (see comments below)
415bool is_crc_valid(uint16_t value, uint8_t crc)
416{
417 // line the bits representing the input in a row (first data, then crc)
418 uint32_t row = (uint32_t)value << 8;
419 row |= crc;
420
421 // polynomial = x^8 + x^5 + x^4 + 1
422 // padded with zeroes corresponding to the bit length of the CRC
423 uint32_t divisor = (uint32_t)0x988000;
424
425 for (int i = 0 ; i < 16 ; i++) {
426
427 // if the input bit above the leftmost divisor bit is 1,
428 // the divisor is XORed into the input
429 if (row & (uint32_t)1 << (23 - i)) {
430 row ^= divisor;
431 }
432
433 // the divisor is then shifted one bit to the right
434 divisor >>= 1;
435 }
436
437 // the remainder should equal zero if there are no detectable errors
438 return (row == 0);
439}
